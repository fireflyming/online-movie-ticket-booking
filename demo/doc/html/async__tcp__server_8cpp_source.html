21class session
22 : public std::enable_shared_from_this<session>
23{
24public:
25 session(tcp::socket socket)
26 : socket_(std::move(socket))
27 {
28 }
29
30 void start()
31 {
32 do_read();
33 }
34
35private:
36 void do_read()
37 {
38 auto self(shared_from_this());
39 socket_.async_read_some(boost::asio::buffer(data_, max_length),
40 [this, self](boost::system::error_code ec, std::size_t length)
41 {
42 if (!ec)
43 {
44 strcpy(data_, "welcome online movie booking service!\r\n");
45 length = strlen(data_);
46 do_write(length);
47 }
48 });
49 }
50
51 void do_write(std::size_t length)
52 {
53 auto self(shared_from_this());
54 boost::asio::async_write(socket_, boost::asio::buffer(data_, length),
55 [this, self](boost::system::error_code ec, std::size_t /*length*/)
56 {
57 if (!ec)
58 {
59 do_read();
60 }
61 });
62 }
63
64 tcp::socket socket_;
65 enum { max_length = 4096 };
66 char data_[max_length];
67};

69class server
70{
71public:
72 server(boost::asio::io_context& io_context, short port)
73 : acceptor_(io_context, tcp::endpoint(tcp::v4(), port))
74 {
75 do_accept();
76 }
77
78private:
79 void do_accept()
80 {
81 acceptor_.async_accept(
82 [this](boost::system::error_code ec, tcp::socket socket)
83 {
84 if (!ec)
85 {
86 std::make_shared<session>(std::move(socket))->start();
87 }
88
89 do_accept();
90 });
91 }
92
93 tcp::acceptor acceptor_;
94};
